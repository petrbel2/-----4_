


DynamicArray(int size) : size(size) {
        if (size < 0) {
            throw std::invalid_argument("Size cannot be negative");
        }
        if (size == 0) { data = nullptr;}
        else {data = new T[size];}
    }

    DynamicArray(T* items, int count) : size(count) {
        if (count < 0) throw std::invalid_argument("Count cannot be negative");
        if (count > 0 && items == nullptr) throw std::invalid_argument("Items pointer must not be null");
        if (count == 0) { data = nullptr;}
        else {data = new T[count];}
        for (int i = 0, i < count, i++) {
            data[i] = items[i];
        }
    }
